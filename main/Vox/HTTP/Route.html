def initialize(verb, key, **params)
  @verb = verb.downcase.to_sym
  @key = key
  @params = params
  @rl_key = "#{@verb}:#{@key}:#{major_param}"
end

def format
  return @key if @params.empty?

  Kernel.format(@key, @params) if @params.any?
end
